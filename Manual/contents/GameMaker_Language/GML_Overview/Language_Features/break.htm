Pause

Die Anweisung break wird verwendet, um ein vorzeitiges Ende einer for, repeat, while oder do / until loop vorzeitig zu beenden, oder um eine switch Anweisung zu sagen, dass sie an dieser Stelle enden soll, oder um einen with Aufruf zu beenden. Bitte lesen Sie die einzelnen Seiten für diese verschiedenen Funktionen, um eine genauere Erklärung zu erhalten, wie sie unter den jeweiligen Umständen verwendet werden können. Beachten Sie, dass die Verwendung von break außerhalb eines dieser Kontexte zu einem Fehler führt.

Nachfolgend sehen Sie einige grundlegende Beispiele, wie break verwendet werden kann, und seine syntax ist einfach:

break;

break in einer for Schleife:

for (var i = 0; i < 10; i += 1)
{
    if (array[i] == 234)
    {
        break;
    }
}

num = i;

break in einer repeat Schleife:

var i = 0;
var temp = 0;
repeat (10)
{
    temp += array[i];
    if (temp > max_total)
    {
        break;
    }
    else
    {
        i += 1;
    }
}

break in einer while Schleife:

var i = 0;
while (!place_free(x, y))
{
    x = random(room_width);
    y = random(room_height);
    if (i > 50)
    {
        break;
    }
    else
    {
        i+=1;
    }
}

break in einer do / until Schleife:

var _id = noone;
do
{
    _id = list[| 0];
    if instance_exists(_id)
    {
        break;
    }
    ds_list_delete(list, 0);
}
until (ds_list_empty(list));
target = _id;

break bei Verwendung von with:

var count = 0;
with (obj_Enemy)
{
    count++;
    if (count > 10)
    {
        break;
    }
    hp = 100;
}

break in einer switch:

switch (keyboard_key)
{
    case vk_left:
    case ord("A"):
        x -= 4;
    break;

    case vk_right:
    case ord("D"):
        x += 4;
    break;

    case vk_up:
    case ord("W"):
        y -= 4;
    break;

    case vk_down:
    case ord("S"):
        y += 4;
    break;
}
